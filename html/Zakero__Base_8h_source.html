650 inline std::string join(std::vector<std::string> vector_string
651 , const std::string_view delimiter
652 ) noexcept
653 {
654 if(vector_string.empty())
655 {
656 return {};

658
659 size_t len = 0;

661 {
662 len += str.size() + 1;
663 }
664
665 std::string retval = {};
666 retval.reserve(len);
667
668 for(const std::string& str : vector_string)

698 [[nodiscard]]
699 inline std::vector<std::string> split(const std::string_view string
700 , const std::string_view delimiter
701 ) noexcept
702 {
703 if(string.empty())
704 {
705 return {};
706 }
707
708 std::vector<std::string> retval = {};
709
710 std::string::size_type start = 0;
711 std::string::size_type end = 0;
712
713 while(true)
714 {
715 end = string.find(delimiter, start);
716
717 if(end == std::string::npos)
718 {
719 retval.emplace_back(string.substr(start, string.size()));
720 break;
721 }
722
723 retval.emplace_back(string.substr(start, end - start));
724 start = end + delimiter.size();
725 }
726
727 return retval;
728 }
